\begin{notedef}
  \texttt{CompositeIntVarSelector}$(h_1,h_2)$ selects a constraint according to heuristic $h_1$, then selects an integer variable involved in the constraint according to heuristic $h_2$:
$$h_2(support(h_1))$$
  \end{notedef}
